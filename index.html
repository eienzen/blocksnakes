} function referUser(address referrer, address referee) internal { if (hasBeenReferred[referee] || referee == referrer || referee == address(0) || referrer == address(0)) { return; // डुप्लिकेशन रोकने के लिए } referredBy[referee] = referrer; hasBeenReferred[referee] = true; playerHistory[referrer].totalReferrals += 1; } function addReferralReward(address user, uint256 reward, address referee) internal { if (reward == 0) return; uint256 rewardInWei = reward * 10 ** 18; balanceOf[user] += rewardInWei; playerHistory[user].referralRewards += rewardInWei; playerHistory[user].totalRewards += rewardInWei; playerHistory[user].rewardHistory.push(RewardEntry({ amount: rewardInWei, timestamp: block.timestamp, rewardType: "Referral", referee: referee })); emit ReferralAdded(user, referee, rewardInWei); } function claimAllRewards( uint256 totalReward, address referrer, address referee, uint256 referrerReward, uint256 refereeReward ) external { // गेम रिवॉर्ड्स जोड़ें if (totalReward > 0) { uint256 totalRewardInWei = totalReward * 10 ** 18; balanceOf[msg.sender] += totalRewardInWei; playerHistory[msg.sender].totalRewards += totalRewardInWei; playerHistory[msg.sender].rewardHistory.push(RewardEntry({ amount: totalRewardInWei, timestamp: block.timestamp, rewardType: "Claim", referee: address(0) })); emit RewardsAdded(msg.sender, totalRewardInWei, "Claim"); } // रेफरल रजिस्ट्रेशन और रिवॉर्ड्स if (referrer != address(0) && referee != address(0)) { referUser(referrer, referee); if (referrerReward > 0) { addReferralReward(referrer, referrerReward, referee); } if (refereeReward > 0) { addReferralReward(referee, refereeReward, address(0)); } } } function getRewardHistory(address user) external view returns (RewardEntry[] memory) { return playerHistory[user].rewardHistory; emit RewardClaimed(msg.sender, reward); } function unstakeTokens() external { uint256 amount = stakedBalance[msg.sender]; require(amount > 0, "No tokens to unstake"); stakedBalance[msg.sender] = 0; balanceOf[msg.sender] += amount; emit Unstaked(msg.sender, amount); } }
